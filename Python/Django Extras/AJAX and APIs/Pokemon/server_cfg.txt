[Unit]
Description=gunicorn daemon
After=network.target
[Service]
User=ubuntu
Group=www-data
WorkingDirectory=/home/ubuntu/django_pokedex
ExecStart=/home/ubuntu/django_pokedex/env/bin/gunicorn --workers 3 --bind unix:/home/ubuntu/django_pokedex/pokemon.sock pokemon.wsgi:application
[Install]
WantedBy=multi-user.target


server {
  listen 80;
  server_name 3.133.152.89;
  location = /favicon.ico { access_log off; log_not_found off; }
  location /static/ {
      root /home/ubuntu/django-pokedex;
  }
  location / {
      include proxy_params;
      proxy_pass http://unix:/home/ubuntu/django-pokedex/pokemon.sock;
  }
}
